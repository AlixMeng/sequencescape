def completed_view
  @wells.dup.tap do |wells|
    complete(wells)
  end.each_with_index.inject([]) do |wells, (well, index)|
    wells.tap { wells[@shape.horizontal_to_vertical(index + 1, @size)] = well }
  end.compact
end
