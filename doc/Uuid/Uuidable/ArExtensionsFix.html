def records_for_missing_uuids(&block)
  self.connection.select_all(%Q{
    SELECT r.id AS id
    FROM #{self.quoted_table_name} r
    LEFT OUTER JOIN #{Uuid.quoted_table_name} u
    ON r.id=u.resource_id AND u.resource_type="#{self}"
    WHERE u.id IS NULL
  }).map { |r| yield(r['id']) }
end
