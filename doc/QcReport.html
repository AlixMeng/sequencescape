def generate_report_identifier
  return true if study.nil? || product_criteria.nil?
  rid = [
    study.abbreviation,
    product_criteria.product.name,
    DateTime.now.to_formatted_s(:number)
  ].compact.join('_').downcase.gsub(/[^\w]/, '_')
  self.report_identifier = rid
end
